Dijkstra's Algorithm(Non-negative weights)

this algorithm is used to find the Single-source shortest path from a specified node 
when all edge weighs are non-negative 
so it can not work with negative edge weights

so the weight is from the current to the initial one

//time complexity: O((V+E)logV)
Dijkstra(G,int s,int V){
    vector<bool> vis(v,False)
    vector<long long> D(v,le18) //consider distances as infinity
    priprity_queue<pair<long long,int>,
                    vector<pair<long long,int>>,
                    greater(pair<long long,int>)
    D[s]=0;//initilise the distance fot the inital code as zero
    pq.push({0,s})

    
    while(!pq.empty()){ //O(V)
        auto& w,u=pq.front(); //O(1)
        pq.pop();   //O(log V)
        if(vis[u])continue;
        vis[u]=true;

        for(auto& e:G[u]){  //O(E)
            nei_w = e.first;
            nei = e.second;
            if(!vis[nei] && D[nei]>D[u]+ nei_w){
                D[nei]=D[u]+nei_w;
                pq.push({D[nei_w],nei}) //O(log V)
            }

        }
    }
}
